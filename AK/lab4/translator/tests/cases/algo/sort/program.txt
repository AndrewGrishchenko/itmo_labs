int[] arr = in();
int n = arr.size();
int i = 0;
int j = 0;
int temp = 0;

while (i < n) {
    j = 0;
    while (j < n - i - 1) {
        if (arr[j] > arr[j+1]) {
            temp = arr[j];
            arr[j] = arr[j+1];
            arr[j+1] = temp;
        }
        j = j + 1;
    }
    i = i + 1;
}

out(arr);
